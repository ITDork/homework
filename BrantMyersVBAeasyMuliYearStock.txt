tickerVol - a variable to hold volume As Integer'tickerTotal - a variable to hold sum of tickerVol'ticker - the stock identifer as String'iRow - as integerSub tickerVolTot()Dim ws As WorksheetDim tickerVol As LongDim tickerTotal As DoubleDim ticker As StringDim iTicker As IntegerDim iRow As LongDim lastRow As LongFor lastRow = 2 To 80000        If (Cells(lastRow + 1, 1).Value = "") Then            Exit For        End If    Next lastRowiTicker = 2For iRow = 2 To lastRow        If Cells(iRow, 1).Value = Cells((iRow + 1), 1).Value Then    tickerVol = Cells(iRow, 7).Value    tickerTotal = tickerTotal + tickerVol    ticker = Cells(iRow, 1).Value    Else    tickerVol = Cells(iRow, 7).Value    tickerTotal = tickerVol + tickerTotal    ticker = Cells(iRow, 1).Value    Cells(iTicker, 9).Value = ticker    Cells(iTicker, 10).Value = tickerTotal    iTicker = iTicker + 1    tickerTotal = 0        End IfNext iRowEnd Sub
